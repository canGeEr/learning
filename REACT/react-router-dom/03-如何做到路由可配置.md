# 手写React可配置路由
- renderProps或者render属性
- layout级别需要使用children属性渲染子路由，不然无法实现层次路由

```javascript
import React from 'react';
import {
  BrowserRouter as Router,
  Switch,
  Route,
} from 'react-router-dom';

class ReactRouter {
  router = null;

  static NoMatch = () => (
    <p>
      没有匹配到任何东西诶，宝~
    </p>
  )

  static renderRouter(router, staticPath) {
    if (!router) return null;
    console.log(router.path);
    return (
      <Switch>
        {router.map((route) => {
          const Component = route.component;
          const children = this.renderRouter(route.children, route.path);
          return (
            <Route
              key={route.name}
              render={(props) => (
                <Component {...props}>
                  {children}
                </Component>
              )}
              path={route.path}
            />
          );
        })}
        <Route path={`${staticPath}/:name+`} component={ReactRouter.NoMatch} />
      </Switch>
    );
  }

  constructor(router) {
    this.router = router;
  }

  render() {
    return (
      <Router>
        {ReactRouter.renderRouter(this.router, '/')}
      </Router>
    );
  }
}

export default ReactRouter;
```
