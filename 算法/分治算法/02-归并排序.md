## 合并有序数组

```javascript
function mergeOrderArr(arr1, arr2, bool) {
  const arr3 = [];
  let arr1Index = 0,
    arr2Index = 0;
  while (arr1Index < arr1.length && arr2Index < arr2.length) {
    arr3.push(
      arr1[arr1Index] > arr2[arr2Index] ? arr2[arr2Index++] : arr1[arr1Index++]
    );
  }
  while (arr1Index < arr1.length) {
    arr3.push(arr1[arr1Index++]);
  }
  while (arr2Index < arr2.length) {
    arr3.push(arr2[arr2Index++]);
  }
  return arr3;
}
```

## 归并排序

```javascript
//使用递归，核心就是将数组一直二分下去
function mergeArr(arr, left, right) {
  if (left >= right) return;
  let arr3 = new Array(right - left + 1);
  let middle = Math.floor((left + right) / 2);
  mergeArr(arr, left, middle);
  mergeArr(arr, middle + 1, right);
  let arr1Index, arr2Index, arr3Index;
  arr1Index = left;
  arr2Index = middle + 1;
  arr3Index = 0;
  while (arr1Index <= middle && arr2Index <= right) {
    arr3[arr3Index++] =
      arr[arr1Index] < arr[arr2Index] ? arr[arr1Index++] : arr[arr2Index++];
  }
  while (arr1Index <= middle) {
    arr3[arr3Index++] = arr[arr1Index++];
  }
  while (arr2Index <= right) {
    arr3[arr3Index++] = arr[arr2Index++];
  }
  for (let i = 0, len = right - left + 1; i < len; i++) {
    arr[left + i] = arr3[i];
  }
}

function mergeSort(arr, left, right) {
  if (right <= left) return [arr[left]];
  const middle = Math.floor((right + left) / 2);
  const leftOrderArr = mergeSort(arr, left, middle);
  const rightOrderArr = mergeSort(arr, middle + 1, right);
  return mergeOrderArr(leftOrderArr, rightOrderArr);
}
```
