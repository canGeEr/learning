## 柯里化

```javascript
function curry(callback, ...args) {
  if(args.length >= callback.length) return callback.apply(null, args.slice(0, callback.length))

  return function(...innerArgs) {
    return curry(callback, ...args, ...innerArgs)
  }
}

const callback = function (a, b, c) {
  return a + b + c;
};

console.log(curry(callback)(1)(2)(3));
```

## 函数组合

```javascript
// 从右往左
function combination(...args) {
  return function (initValue) {
    args.reverse();
    return args.reduce((last, current) => {
      return current(last);
    }, initValue);
  };
}

function add(x) {
  return x + 2;
}

function multi(x) {
  return x * 2;
}

console.log(combination(multi, add)(1));
```
