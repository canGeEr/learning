## 柯里化

```javascript
function curry(callback, context) {
  let args = [];

  return function curried(...currentAgs) {
    if (args.length + currentAgs.length >= callback.length) {
      args = [...args, ...currentAgs.slice(0, callback.length - args.length)];
      return callback.apply(context, args);
    } else {
      args = [...args, ...currentAgs];
      return curried;
    }
  };
}
```

## 函数组合

```javascript
function combination(...args) {
  // 参数是一系列函数，有点像中间件其实
  return function () {
    const initValue = args.shift()(...arguments);
    return args.reduce((last, current) => {
      return current(last);
    }, initValue);
  };
}
```
