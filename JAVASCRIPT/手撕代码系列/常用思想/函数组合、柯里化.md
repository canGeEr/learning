## 柯里化

```javascript
function curry(callback, ...args) {
  if (callback.length > args.length) {
    return function (...argInner) {
      return curry(callback, ...args, ...argInner);
    };
  } else {
    return callback(...args);
  }
}

const callback = function (a, b, c) {
  return a + b + c;
};

console.log(curry(callback)(1)(2)(3));
```

## 函数组合

```javascript
// 从右往左
function combination(...args) {
  return function (initValue) {
    args.reverse();
    return args.reduce((last, current) => {
      return current(last);
    }, initValue);
  };
}

function add(x) {
  return x + 2;
}

function multi(x) {
  return x * 2;
}

console.log(combination(multi, add)(1));
```
