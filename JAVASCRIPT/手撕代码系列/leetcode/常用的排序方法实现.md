## 冒泡排序

```javascript
function bubbleSort(arr) {
  let length = arr.length;
  let temp = null;
  let i = length - 1;
  while (i) {
    //多少趟
    for (let j = 0; j < i; j++) {
      if (arr[j] > arr[j + 1]) {
        temp = arr[j];
        arr[j] = arr[j + 1];
        arr[j + 1] = temp;
      }
    }
    i--;
  }
  return arr;
}
```

## 选择排序

```javascript
//第一种 生成原数组
function selectionSort(arr) {
  let len = arr.length;
  let i = len - 1;
  let minIndex = null;
  let temp = null;
  while (i) {
    minIndex = 0;
    for (let j = 0; j <= i; j++) {
      if (arr[minIndex] > arr[j]) {
        minIndex = j;
      }
    }
    temp = arr[i];
    arr[i] = arr[minIndex];
    arr[minIndex] = temp;
    i--;
  }
  return arr;
}
```
