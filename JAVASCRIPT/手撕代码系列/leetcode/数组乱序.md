// 数组乱序
// 长度为10 的数组中，随机取两个

// 1. 10随机取1个，9个随机取一个

// 从 start~end 随机获取一个值 包括 start、end
function getRoundRandom(start, end) {
  return start + Math.floor(Math.random() * (end - start + 1))
}

// 随机选取
function shuttle(arr, count) {  
  let start = 0
  let newArr = []
  while(start < count) {
    const randomIndex = getRoundRandom(0, arr.length - 1)
    newArr.push(arr.splice(randomIndex, 1)[0])
    start++;
  }
  return newArr
}

// 随机选取加交换
function shuttle(arr, count) {  
  let start = 0
  while(start < count) {
    const randomIndex = getRoundRandom(start, arr.length - 1)
    swap(arr, start, randomIndex)
    start++;
  }
  return arr.slice(0, count)
}

// 两两随机交换
function shuttle(arr, count) {  
  return arr.sort(()=>(0.5-Math.random())).slice(0, count)
}

function swap(arr, i, j) {
  let temp = arr[i]
  arr[i] = arr[j]
  arr[j] = temp
}
